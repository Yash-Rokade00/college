public class BinarySearchExample {

    // Method for Binary Search
    public static Integer binarySearch(Integer[] arr, Integer target) {

        Integer left = 0;
        Integer right = arr.size() - 1;

        while (left <= right) {

            // Find middle index
            Integer mid = (left + right) / 2;

            // If target found
            if (arr[mid] == target) {
                return mid;
            }

            // If target is greater, search right half
            else if (arr[mid] < target) {
                left = mid + 1;
            }

            // If target is smaller, search left half
            else {
                right = mid - 1;
            }
        }

        // Target not found
        return -1;
    }

    // Main method
    public static void mainMethod() {

        Integer[] numbers = new Integer[]{10, 20, 30, 40, 50, 60, 70};

        Integer target = 50;

        Integer result = binarySearch(numbers, target);

        if (result != -1) {
            System.debug('Element found at index: ' + result);
        } else {
            System.debug('Element not found');
        }
    }
}
